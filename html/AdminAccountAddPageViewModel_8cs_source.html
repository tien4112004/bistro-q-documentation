152 public async Task AddAccountAsync()
153 {
154 Form.ValidateAll();
155 if (Form.HasErrors)
156 {
157 await _dialogService.ShowErrorDialog("Data is invalid. Please check again.", "Error");
158 return;
159 }
160
161 try
162 {
163 IsProcessing = true;
164
165 var request = new CreateAccountRequest
166 {
167 Username = Form.Username,
168 Password = Form.Password,
169 Role = Form.Role,
170 TableId = Form.TableId
171 };
172
173 await _accountDataService.CreateAccountAsync(request);
174
175 await _dialogService.ShowSuccessDialog("Account added successfully.", "Success");
176 NavigateBack();
177 }
178 catch (Exception ex)
179 {
180 await _dialogService.ShowErrorDialog(ex.Message, "Error");
181 }
182 finally
183 {
184 IsProcessing = false;
185 }
