189 public async Task UpdateAccountAsync()
190 {
191 ValidateForm();
192 if (!CanEditAccount())
193 {
194 await _dialogService.ShowErrorDialog("Data is invalid. Please check again.", "Error");
195 return;
196 }
197
198 try
199 {
200 IsProcessing = true;
201
202 var request = new UpdateAccountRequest
203 {
204 Username = Form.Username,
205 Role = Form.Role,
206 Password = Form.Password,
207 TableId = Form.TableId
208 };
209
210 await _accountDataService.UpdateAccountAsync(Account.UserId, request);
211
212 await _dialogService.ShowSuccessDialog("Account updated successfully.", "Success");
213 NavigateBack();
214 }
215 catch (Exception ex)
216 {
217 Debug.WriteLine(ex.StackTrace);
218 await _dialogService.ShowErrorDialog(ex.Message, "Error");
219 }
220 finally
221 {
222 IsProcessing = false;
223 }
224 }
