174 private async Task LoadDataAsync()
175 {
176 if (State.IsLoading || _isDisposed) return;
177 try
178 {
179 State.IsLoading = true;
180
181 var result = await _accountDataService.GetAccountsAsync(State.Query);
182
183 var accounts = _mapper.Map<IEnumerable<AccountViewModel>>(result.Data);
184 State.Source = new ObservableCollection<AccountViewModel>(accounts);
185 _messenger.Send(new PaginationChangedMessage(
186 result.TotalItems,
187 result.CurrentPage,
188 result.TotalPages
189 ));
190 }
191 catch (Exception ex)
192 {
193 await _dialogService.ShowErrorDialog(ex.Message, "Error");
194 }
195 finally
196 {
197 State.IsLoading = false;
198 }
199 }

204 private async Task DeleteSelectedAccountAsync()
205 {
206 if (State.SelectedAccount?.UserId == null) return;
207
208 try
209 {
210 var result = await _dialogService.ShowConfirmDeleteDialog();
211 if (result != ContentDialogResult.Primary) return;
212
213 var success = await _accountDataService.DeleteAccountAsync(State.SelectedAccount.UserId);
214 if (success)
215 {
216 await _dialogService.ShowSuccessDialog("Account deleted successfully.", "Success");
217 State.Source.Remove(State.SelectedAccount);
218 State.SelectedAccount = null;
219 await LoadDataAsync();
220 }
221 else
222 {
223 await _dialogService.ShowErrorDialog("Failed to delete account.", "Error");
224 }
225 }
226 catch (Exception ex)
227 {
228 await _dialogService.ShowErrorDialog(ex.Message, "Error");
229 }
230 }
