166 private async Task LoadDataAsync()
167 {
168 if (State.IsLoading || _isDisposed) return;
169
170 try
171 {
172 State.IsLoading = true;
173 var result = await _categoryDataService.GetCategoriesAsync(State.Query);
174
175 var categories = _mapper.Map<IEnumerable<CategoryViewModel>>(result.Data);
176 State.Source = new ObservableCollection<CategoryViewModel>(categories);
177 _messenger.Send(new PaginationChangedMessage(
178 result.TotalItems,
179 result.CurrentPage,
180 result.TotalPages
181 ));
182 }
183 catch (Exception ex)
184 {
185 await _dialogService.ShowErrorDialog(ex.Message, "Error");
186 }
187 finally
188 {
189 State.IsLoading = false;
190 }
191 }
192
196 private async Task DeleteSelectedCategoryAsync()
197 {
198 if (State.SelectedCategory?.CategoryId == null) return;
199
200 try
201 {
202 var result = await _dialogService.ShowConfirmDeleteDialog();
203 if (result != ContentDialogResult.Primary) return;
204
205 var success = await _categoryDataService.DeleteCategoryAsync(State.SelectedCategory.CategoryId.Value);
206 if (success)
207 {
208 await _dialogService.ShowSuccessDialog("Category deleted successfully.", "Success");
209 State.Source.Remove(State.SelectedCategory);
210 State.SelectedCategory = null;
211 await LoadDataAsync();
212 }
213 else
214 {
215 await _dialogService.ShowErrorDialog("Failed to delete category.", "Error");
216 }
217 }
218 catch (Exception ex)
219 {
220 await _dialogService.ShowErrorDialog(ex.Message, "Error");
221 }
222 }
