98 public async Task UpdateProductAsync()
99 {
100 Form.ValidateAll();
101 if (!CanUpdate())
102 {
103 await _dialogService.ShowErrorDialog("Data is invalid. Please check again.", "Error");
104 return;
105 }
106
107 try
108 {
109 IsProcessing = true;
110 ErrorMessage = string.Empty;
111
112 var request = new UpdateProductRequest
113 {
114 Name = Form.Name,
115 Price = Form.Price,
116 Unit = Form.Unit,
117 DiscountPrice = Form.DiscountPrice,
118 CategoryId = Form.CategoryId,
119 Calories = (decimal)Form.Calories,
120 Fat = (decimal)Form.Fat,
121 Fiber = (decimal)Form.Fiber,
122 Protein = (decimal)Form.Protein,
123 Carbohydrates = (decimal)Form.Carbohydrates,
124 };
125
126 if (Form.ImageFile != null)
127 {
128 await _productDataService.UpdateProductImageAsync(Form.ProductId.Value, Form.ImageFile.Stream, Form.ImageFile.FileName, Form.ImageFile.ContentType);
129 }
130 await _productDataService.UpdateProductAsync(Form.ProductId.Value, request);
131
132 await _dialogService.ShowSuccessDialog($"Successfully updated product: {request.Name}", "Success");
133 NavigateBack();
134 }
135 catch (Exception ex)
136 {
137 ErrorMessage = ex.Message;
138 await _dialogService.ShowErrorDialog(ex.Message, "Error");
139 }
140 finally
141 {
142 IsProcessing = false;
143 }
144 }
