172 private async Task LoadDataAsync()
173 {
174 if (State.IsLoading || _isDisposed) return;
175 try
176 {
177 State.IsLoading = true;
178 var result = await _productDataService.GetProductsAsync(State.Query);
179
180 var products = _mapper.Map<IEnumerable<ProductViewModel>>(result.Data);
181 State.Source = new ObservableCollection<ProductViewModel>(products);
182 _messenger.Send(new PaginationChangedMessage(
183 result.TotalItems,
184 result.CurrentPage,
185 result.TotalPages
186 ));
187 }
188 catch (Exception ex)
189 {
190 Debug.WriteLine(ex.StackTrace);
191 await _dialogService.ShowErrorDialog(ex.Message, "Error");
192 }
193 finally
194 {
195 State.IsLoading = false;
196 }
197 }
198
203 private async Task DeleteSelectedProductAsync()
204 {
205 if (State.SelectedProduct?.ProductId == null) return;
206
207 try
208 {
209 var result = await _dialogService.ShowConfirmDeleteDialog();
210 if (result != ContentDialogResult.Primary) return;
211
212 var success = await _productDataService.DeleteProductAsync(State.SelectedProduct.ProductId.Value);
213 if (success)
214 {
215 await _dialogService.ShowSuccessDialog("Product deleted successfully.", "Success");
216 State.Source.Remove(State.SelectedProduct);
217 State.SelectedProduct = null;
218 await LoadDataAsync();
219 }
220 else
221 {
222 await _dialogService.ShowErrorDialog("Failed to delete product.", "Error");
223 }
224 }
225 catch (Exception ex)
226 {
227 await _dialogService.ShowErrorDialog(ex.Message, "Error");
228 }
229 }
