100 public async Task AddTableAsync()
101 {
102 try
103 {
104 IsProcessing = true;
105 var request = new CreateTableRequest
106 {
107 ZoneId = Form.ZoneId,
108 SeatsCount = Form.SeatsCount
109 };
110
111 if (request.ZoneId == 0)
112 {
113 throw new InvalidDataException("Zone must be selected.");
114 }
115 if (request.SeatsCount == null)
116 {
117 throw new InvalidDataException("Seats count must be greater than 0.");
118 }
119
120 await _tableDataService.CreateTableAsync(request);
121 await _dialogService.ShowSuccessDialog("Table added successfully.", "Success");
122 NavigateBack();
123 }
124 catch (Exception ex)
125 {
126 await _dialogService.ShowErrorDialog(ex.Message, "Error");
127 }
128 finally
129 {
130 IsProcessing = false;
131 }
