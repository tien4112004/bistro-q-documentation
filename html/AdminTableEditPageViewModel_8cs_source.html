95 public async Task UpdateTableAsync()
96 {
97 try
98 {
99 IsProcessing = true;
100
101 if (Request.ZoneId == null)
102 {
103 throw new InvalidDataException("Please choose a zone.");
104 }
105 if (Request.SeatsCount == null)
106 {
107 throw new InvalidDataException("Seats count must be greater than 0.");
108 }
109
110 await _tableDataService.UpdateTableAsync(Table.TableId.Value, Request);
111
112 await _dialogService.ShowSuccessDialog("Table updated successfully.", "Success");
113 NavigateBack();
114 }
115 catch (Exception ex)
116 {
117 await _dialogService.ShowErrorDialog(ex.Message, "Error");
118 }
119 finally
120 {
121 IsProcessing = false;
122 }
123 }
