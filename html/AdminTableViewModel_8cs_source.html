169 private async Task LoadDataAsync()
170 {
171 if (State.IsLoading || _isDisposed) return;
172
173 try
174 {
175 State.IsLoading = true;
176
177 var result = await _tableDataService.GetGridDataAsync(State.Query);
178
179 var tables = _mapper.Map<IEnumerable<TableViewModel>>(result.Data);
180 State.Source = new ObservableCollection<TableViewModel>(tables);
181 _messenger.Send(new PaginationChangedMessage(
182 result.TotalItems,
183 result.CurrentPage,
184 result.TotalPages
185 ));
186 }
187 catch (Exception ex)
188 {
189 await _dialogService.ShowErrorDialog(ex.Message, "Error");
190 }
191 finally
192 {
193 State.IsLoading = false;
194 }
195 }
196
201 private async Task DeleteSelectedTableAsync()
202 {
203 if (State.SelectedTable?.TableId == null) return;
204
205 try
206 {
207 var result = await _dialogService.ShowConfirmDeleteDialog();
208 if (result != ContentDialogResult.Primary) return;
209
210 var success = await _tableDataService.DeleteTableAsync(State.SelectedTable.TableId.Value);
211 if (success)
212 {
213 await _dialogService.ShowSuccessDialog("Table deleted successfully.", "Success");
214 State.Source.Remove(State.SelectedTable);
215 State.SelectedTable = null;
216 await LoadDataAsync();
217 }
218 else
219 {
220 await _dialogService.ShowErrorDialog("Failed to delete table.", "Error");
221 }
222 }
223 catch (Exception ex)
224 {
225 await _dialogService.ShowErrorDialog(ex.Message, "Error");
226 }
227 }

257 public void AdminTableDataGrid_Sorting(object sender, DataGridColumnEventArgs e)
258 {
259 var dataGrid = sender as DataGrid;
260 var column = e.Column;
261 var sortDirection = column.SortDirection == null || column.SortDirection == DataGridSortDirection.Descending
262 ? "asc"
263 : "des";
264
265 foreach (var col in dataGrid.Columns)
266 {
267 if (col != column)
268 {
269 col.SortDirection = null;
270 }
271 }
272
273 column.SortDirection = sortDirection == "asc"
274 ? DataGridSortDirection.Ascending
275 : DataGridSortDirection.Descending;
276
277 var sortParams = (column.Tag.ToString(), sortDirection);
278 SortCommand.Execute(sortParams);
279 }
