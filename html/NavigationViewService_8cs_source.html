99 private NavigationViewItem? GetSelectedItem(IEnumerable<object> menuItems, Type pageType)
100 {
101 foreach (var item in menuItems.OfType<NavigationViewItem>())
102 {
103 if (IsMenuItemForPageType(item, pageType))
104 {
105 return item;
106 }
107
108 var selectedChild = GetSelectedItem(item.MenuItems, pageType);
109 if (selectedChild != null)
110 {
111 return selectedChild;
112 }
113 }
114
115 return null;
116 }

118 private bool IsMenuItemForPageType(NavigationViewItem menuItem, Type sourcePageType)

123 }
