232 public void Receive(AddProductToCartMessage message)
233 {
234 var product = message.Product;
235 var existingOrderItem = CartItems.FirstOrDefault(od => od.ProductId == product.ProductId);
236 if (existingOrderItem != null)
237 {
238 existingOrderItem.Quantity++;
239 }
240 else
241 {
242 CartItems.Add(new OrderItemViewModel
243 {
244 OrderId = Order.OrderId,
245 ProductId = product.ProductId,
246 Product = product,
247 Quantity = 1,
248 PriceAtPurchase = (product.DiscountPrice != 0 ? product.DiscountPrice : product.Price)
249 });
250 }
251 }
252
257 public async void Receive(OrderRequestedMessage message)
258 {
259 try
260 {

262
263 await _orderDataService.CreateOrderItems(cart);
264 dispatcherQueue.TryEnqueue(() =>

266 CartItems.Clear();
267 _messenger.Send(new OrderSucceededMessage());
268 });

377 private void SeparateOrdersByStatus()
378 {
379 ProcessingItems.Clear();
380 CompletedItems.Clear();
381
382 foreach (var item in Order.OrderItems)
383 {
384 if (item.Status == OrderItemStatus.InProgress || item.Status == OrderItemStatus.Pending)
385 {
386 ProcessingItems.Add(item);
387 }
388 else if (item.Status == OrderItemStatus.Completed)
389 {
390 CompletedItems.Add(item);
391 }
392 }
393
394 OnPropertyChanged(nameof(IsProcessingItemsEmpty));
395 OnPropertyChanged(nameof(IsCompletedItemsEmpty));
396 }
