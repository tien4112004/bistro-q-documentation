193 private Task<string> EncryptData(string data)
194 {
195 if (string.IsNullOrEmpty(data))
196 throw new ArgumentNullException(nameof(data));
197
198 var jsonData = JsonSerializer.Serialize(data);
199 var dataBytes = Encoding.UTF8.GetBytes(jsonData);
200
201 // Encrypt using DPAPI
202#pragma warning disable CA1416 // Validate platform compatibility
203 var encryptedData = ProtectedData.Protect(
204 dataBytes,
205 _entropy,
206 DataProtectionScope.CurrentUser
207 );
208
209 return Task.FromResult(Convert.ToBase64String(encryptedData));
210 }
211
219 private Task<string> DecryptData(string data)
220 {
221 if (string.IsNullOrEmpty(data))
222 throw new ArgumentNullException(nameof(data));
223
224 var byteData = Convert.FromBase64String(data);
225
226 // Decrypt using DPAPI
227#pragma warning disable CA1416 // Validate platform compatibility
228 var decryptedData = ProtectedData.Unprotect(
229 byteData,
230 _entropy,
231 DataProtectionScope.CurrentUser
232 );
233
234 var jsonData = Encoding.UTF8.GetString(decryptedData);
235 var dataString = JsonSerializer.Deserialize<string>(jsonData);
236
237 return Task.FromResult(dataString);
238 }
