81 public void Receive(TableStateChangedMessage message)
82 {
83 if (message?.TableId == null) return;
84
85 var table = ViewModel.Tables.FirstOrDefault(t => t.TableId == message.TableId);
86 if (table == null) return;
87
88 dispatcherQueue.TryEnqueue(() =>
89 {
90 switch (message.State)
91 {
92 case CashierTableState.CheckoutPending:
93 table.IsCheckingOut = true;
94 break;
95 case CashierTableState.Occupied:
96 table.IsOccupied = true;
97 break;
98 case CashierTableState.Available:
99 table.IsOccupied = false;
100 table.IsCheckingOut = false;
101 break;
102 }
103
104 var container = myGridView.ContainerFromItem(table) as GridViewItem;
105 if (container == null) return;
106
107 var styleSelector = (TableItemStyleSelector)Resources["TableItemStyleSelector"];
108 container.Style = styleSelector.SelectStyle(table, container);
109
110 container.UpdateLayout();
111 ViewModel.UpdateZoneState();
112 });
113 }
114}
