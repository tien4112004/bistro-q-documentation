93 public async Task OnZoneChangedAsync(int? zoneId, string type)
94 {
95 IsLoading = true;
96 try
97 {
98 if (zoneId == null)
99 {
100 await Task.Delay(200);
101 IsLoading = false;
102 return;
103 }
104
105 var tablesData = await TaskHelper.WithMinimumDelay(
106 _tableDataService.GetTablesByCashierAsync(zoneId.Value, type),
107 200);
108
109 var tables = _mapper.Map<IEnumerable<TableViewModel>>(tablesData);
110 Tables = new ObservableCollection<TableViewModel>(tables);
111 if (Tables.Any())
112 {
113 SelectedTable = Tables.First();
114 }
115 }
116 catch (Exception ex)
117 {
118 Debug.WriteLine(ex.Message);
119 Tables = new ObservableCollection<TableViewModel>();
120 SelectedTable = null;
121 }
122 finally
123 {
124 IsLoading = false;
125 }
126 }
